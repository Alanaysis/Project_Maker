## 代码片段收集

### 量化相关

```python
# 对称量化
def symmetric_quantize(tensor, num_bits=8):
    """对称量化"""
    alpha = np.max(np.abs(tensor))
    scale = alpha / (2 ** (num_bits - 1) - 1)
    quantized = np.clip(np.round(tensor / scale), -2**(num_bits-1), 2**(num_bits-1)-1)
    return quantized.astype(np.int8), scale, 0
```

```python
# 非对称量化
def asymmetric_quantize(tensor, num_bits=8):
    """非对称量化"""
    min_val = np.min(tensor)
    max_val = np.max(tensor)
    scale = (max_val - min_val) / (2 ** num_bits - 1)
    zero_point = np.round(-min_val / scale)
    quantized = np.clip(np.round(tensor / scale) + zero_point, 0, 2**num_bits-1)
    return quantized.astype(np.uint8), scale, zero_point
```

### 算子融合相关

```python
# Conv+BN 融合
def fuse_conv_bn(conv_weight, conv_bias, bn_mean, bn_var, bn_weight, bn_bias, eps=1e-5):
    """融合 Conv 和 BN"""
    scale = bn_weight / np.sqrt(bn_var + eps)
    fused_weight = conv_weight * scale.reshape(-1, 1, 1, 1)
    fused_bias = (conv_bias - bn_mean) * scale + bn_bias
    return fused_weight, fused_bias
```
